74unsigned int gcd(unsigned int a, unsigned int b)
75{
76 unsigned int t;
77 while (b != 0u) {
78 t = b;
79 b = a % b;
80 a = t;
81 }
82 return a;
83}

87// The result should fit into a unsigned int(e.g. on 32bit system 4294967296).
88//
89// C(15,10) = 3003
90// C(17,10) = 19448
91// C(18,10) = 43758
92// int16 = 65536
93// C(19,10) = 92378
94// C(20,10) = 184756
95// C(30,10) = 30045015
96// int32 = 4294967296
97unsigned int comb_num(unsigned int n, unsigned int r)
98{
99 const unsigned int MAX_N_R = 100;
100
101 if (n == 0 || r == 0) return 1;
102
103 // not allowed, regardless of mathematical meaning.
104 if (n < r) return 0;
105
106 // obviously too large fitting in a unsigned int
107 if (r > MAX_N_R && (n-r) > MAX_N_R) return 0;
108
109 unsigned int k = (r > n-r ? n-r : r);
110
111 std::vector<int> numerator(k), denominator(k);
112
113 for (unsigned int i = 0; i < k; ++i) {
114 numerator[i] = n - i;
115 denominator[i] = k - i;
116 }
117
118 int c = 1;
119 for (size_t i = 0; i < k; ++i) {
120 // deal with denominator[i]
121 for (size_t j = 0; j < k; ++j) {
122 if ( (c=gcd(denominator[i], numerator[j])) > 1 ) {
123 if (numerator[j] % c != 0 || denominator[i] % c != 0) {
124 #ifdef DEBUG
125 std::cerr << "ERROR: gcd error, " << k << " = gcd("
126 << denominator[i] << ","
127 << numerator[j] << ")" << std::endl;
128 #endif
129
130 return 0;
131 }
132 numerator[j] /= c;
133 denominator[i] /= c;
134 }
135 }
136 }
137
138 // check if the result is an integer, i.e. the denominator vector is
139 // (1,1,1, ... ,1)
140 for (size_t i = 0; i < k; ++i) {
141 if (denominator[i] != 1) {
142 #ifdef DEBUG
143 std::cerr << "ERROR: denominator is not dividable by numerator"
144 << std::endl;
145 #endif
146
147 return 0;
148 }
149 }
150
151 size_t N = 1;
152
153 for (size_t i = 0; i < k; ++i) {
154 if (MAX_N_BASE*1.0/numerator[i] < N) {
155 return 0;
156 }
157 N *= numerator[i];
158 }
159 return N;
160}

164void init_order_index(unsigned int nv, unsigned int nd)
165{
166 order_index = new unsigned int[nd+2];
167 size_t i = 0, N = 0;
168 order_index[0] = 0;
169
170 while (i < nd+1) {
171 N += comb_num(nv+i-1, i);
172 order_index[++i] = N;
173 }
174 order_index[nd+1] = FULL_VEC_LEN;
175}

249 //std::cout << "Mem: " << NS << std::endl;

286{

399#ifdef MSVC_DLL
400extern "C" _declspec(dllexport) void _stdcall ad_init(const TNVND* nvar, const TNVND* nord)
401#else
402void ad_init(const TNVND* nvar, const TNVND* nord)
403#endif
404{
405 TNVND nv = *nvar;
406 TNVND nd = *nord;
407
408 #ifdef DEBUG
409 std::cerr << "Initialize nv= " << *nvar << " nd= " << *nord << std::endl;
410 #endif
411 advecpool = NULL;
412 gnv = nv; // assign to global nv
413 gnd = nd; // assign to global nd
414
415 FULL_VEC_LEN = comb_num(gnv+gnd, gnd);
416 init_order_index(gnv, gnd);
417
418 // overflow !!
419 if (FULL_VEC_LEN == 0) {
420 std::cerr << "Overflow! (" << nv << "," << nd << ")" << std::endl;
421 std::exit(1);
422 }
423
424 // this reset the base vector
425 vec = new TNVND[nv];
426 for (TNVND i = 0; i < nv; ++i) vec[i] = 0;
427
428 nvec = 0;
429
430 init_base(gnv, gnd);
431
432 init_prod_index(gnv, gnd);
433
434 // random number generator
435 srand(1U);
436}
